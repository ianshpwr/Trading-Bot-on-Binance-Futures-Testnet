
Usage / Run Examples: 

# Market BUY
python cli.py place-order --symbol BTCUSDT --side BUY --type MARKET --quantity 0.001

# Limit SELL
python cli.py place-order --symbol BTCUSDT --side SELL --type LIMIT --quantity 0.001 --price 50000

# Stop-Limit BUY
python cli.py place-order --symbol BTCUSDT --side BUY --type STOP --quantity 0.001 --price 45000 --stop-price 44500

# Help
python cli.py --help
